#Advanced Function with Parameters

function Get-DiskInfo {
    [CmdletBinding()]
    param(
        [Parameter(Mandatory)]
        [string]$ComputerName
    )

    Get-WmiObject Win32_LogicalDisk -ComputerName $ComputerName |
        Select-Object DeviceID, Size, FreeSpace
}
